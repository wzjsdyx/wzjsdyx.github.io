### 汉明码的局限性

<font color=blue>汉明码的码距为3，只能保证检测2个错误或纠正1个错误；</font>

<font color=red>码距怎么理解</font>

<font color=blue>其他的错误超出其能力范围，不去考虑；</font>

> 多比特发生错误时，汉明校验子的情况很复杂：
>
> 1、校验子非全0但指向错误位置——常见的情况
>
> 2、发生多bit错误，但是，校验子全为0——概率极低
>
> 3、发生多bit错误，但是，校验子不是全为0，但是指向一个实际错误的地址——概率低

/*
 ECC: SECDED(Single Error Correction, Double Error Detection)
*/
module ecc_a0d64_8(
    // encode
    input       [63:0]  enc_in      ,
    output      [71:0]  enc_out     ,

    // decode
    input       [71:0]  dec_in      ,
    output reg  [63:0]  dec_out     ,
    // control bits
    input               dec_bypass  ,
    input               rd_en       ,
    // error status
    output      [2:0]   ecc_err_stus
);

parameter  P1   = 64'hAB55555556AAAD5B;
parameter  P2   = 64'hCD9999999B33366D;
parameter  P4   = 64'hF1E1E1E1E3C3C78E;
parameter  P8   = 64'h01FE01FE03FC07F0;
parameter  P16  = 64'h01FFFE0003FFF800;
parameter  P32  = 64'h01FFFFFFFC000000;
parameter  P64  = 64'hFE00000000000000;


//***************************************
//          encode 
//***************************************
reg [6:0] hamming_check_code;

always@(*)begin
    hamming_check_code[0] = (^(enc_in & P1 ));
    hamming_check_code[1] = (^(enc_in & P2 ));
    hamming_check_code[2] = (^(enc_in & P4 ));
    hamming_check_code[3] = (^(enc_in & P8 ));
    hamming_check_code[4] = (^(enc_in & P16));
    hamming_check_code[5] = (^(enc_in & P32));
    hamming_check_code[6] = (^(enc_in & P64));
end

assign enc_out[63:0 ]   = enc_in[63:0];
assign enc_out[70:64]   = hamming_check_code[6:0];

assign enc_out[71] = ^(enc_out[70:0]); // overall parity check bit

//***************************************
//          decode 
//***************************************

reg [6:0] hamming_syndrome;
always@(*)begin
    hamming_syndrome[0] = (^{dec_in[64],(dec_in[63:0] & P1 )}) & rd_en & (~dec_bypass);
    hamming_syndrome[1] = (^{dec_in[65],(dec_in[63:0] & P2 )}) & rd_en & (~dec_bypass);
    hamming_syndrome[2] = (^{dec_in[66],(dec_in[63:0] & P4 )}) & rd_en & (~dec_bypass);
    hamming_syndrome[3] = (^{dec_in[67],(dec_in[63:0] & P8 )}) & rd_en & (~dec_bypass);
    hamming_syndrome[4] = (^{dec_in[68],(dec_in[63:0] & P16)}) & rd_en & (~dec_bypass);
    hamming_syndrome[5] = (^{dec_in[69],(dec_in[63:0] & P32)}) & rd_en & (~dec_bypass);
    hamming_syndrome[6] = (^{dec_in[70],(dec_in[63:0] & P64)}) & rd_en & (~dec_bypass);
end

assign parity_check = (^(dec_in[71:0])) & rd_en & (~dec_bypass);

// error status
wire abnormal_status; // Transmit have error ,but exceeding the detection capability of hamming code 
wire [2:0] err_stus;
assign abnormal_status = hamming_syndrome[6:0] > 7'd71; // hamming code just only point H1-H71
assign err_stus[2:0] = {abnormal_status,parity_check,|hamming_syndrome};

// 1 bit correct and decode data out
wire err_1bit;
assign err_1bit = (~abnormal_status) & parity_check & (|hamming_syndrome);
always@(*)begin
    if(err_1bit)begin
        case(hamming_syndrome[6:0])
            7'd3  : dec_out = {dec_in[ 63:  1], ~dec_in[  0]                 };
            7'd5  : dec_out = {dec_in[ 63:  2], ~dec_in[  1], dec_in[  0:  0]};
            7'd6  : dec_out = {dec_in[ 63:  3], ~dec_in[  2], dec_in[  1:  0]};
            7'd7  : dec_out = {dec_in[ 63:  4], ~dec_in[  3], dec_in[  2:  0]};
            7'd9  : dec_out = {dec_in[ 63:  5], ~dec_in[  4], dec_in[  3:  0]};
            7'd10 : dec_out = {dec_in[ 63:  6], ~dec_in[  5], dec_in[  4:  0]};
            7'd11 : dec_out = {dec_in[ 63:  7], ~dec_in[  6], dec_in[  5:  0]};
            7'd12 : dec_out = {dec_in[ 63:  8], ~dec_in[  7], dec_in[  6:  0]};
            7'd13 : dec_out = {dec_in[ 63:  9], ~dec_in[  8], dec_in[  7:  0]};
            7'd14 : dec_out = {dec_in[ 63: 10], ~dec_in[  9], dec_in[  8:  0]};
            7'd15 : dec_out = {dec_in[ 63: 11], ~dec_in[ 10], dec_in[  9:  0]};
            7'd17 : dec_out = {dec_in[ 63: 12], ~dec_in[ 11], dec_in[ 10:  0]};
