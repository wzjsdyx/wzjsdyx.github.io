# 方式1：基本数据类型

```c
#include <stdint.h>  // 包含标准整数类型

void basic_data_types(void)
{
    // Byte (8位) 访问
    uint8_t byte_data = 0xA5;           // 无符号字节
    int8_t signed_byte = -128;          // 有符号字节
    
    // Half word (16位) 访问  
    uint16_t half_word = 0x1234;        // 无符号半字
    int16_t signed_half = -32768;       // 有符号半字
    
    // Word (32位) 访问
    uint32_t word_data = 0x12345678;    // 无符号字
    int32_t signed_word = -2147483648;  // 有符号字
    
    // Double word (64位) 访问
    uint64_t dword_data = 0x123456789ABCDEF0;  // 无符号双字
    int64_t signed_dword = -9223372036854775808; // 有符号双字
}
```



# 方式2：指针访问内存的不同宽度

```c
#include <stdio.h>
#include <stdint.h>

void pointer_access_example(void)
{
    // 假设这是内存地址（在嵌入式系统中可能是物理地址）
    uint8_t *base_address = (uint8_t *)0x1000;
    
    printf("=== 通过指针访问不同宽度数据 ===\n\n");
    
    // 1. Byte 访问
    uint8_t *byte_ptr = base_address;
    printf("Byte 访问:\n");
    printf("  address[0] = 0x%02X\n", byte_ptr[0]);
    printf("  address[1] = 0x%02X\n", byte_ptr[1]);
    
    // 2. Half word (16位) 访问
    // 注意：需要考虑对齐和字节序！
    uint16_t *half_word_ptr = (uint16_t *)base_address;
    printf("\nHalf word 访问:\n");
    printf("  half_word_ptr[0] = 0x%04X\n", half_word_ptr[0]);
    
    // 3. Word (32位) 访问
    uint32_t *word_ptr = (uint32_t *)base_address;
    printf("\nWord 访问:\n");
    printf("  word_ptr[0] = 0x%08X\n", word_ptr[0]);
    
    // 4. Double word (64位) 访问
    uint64_t *dword_ptr = (uint64_t *)base_address;
    printf("\nDouble word 访问:\n");
    printf("  dword_ptr[0] = 0x%016llX\n", dword_ptr[0]);
}
```
