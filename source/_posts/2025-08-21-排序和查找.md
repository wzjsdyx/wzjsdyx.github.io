# 快速排序算法

```c
#include<stdio.h>

// #define LEN 8

// #define LEN 3

// #define LEN 5

#define LEN 7
int a[LEN] = { 5, 2, 4, 7, 1, 3, 2 };

int partition(int start, int end)
{
    int pivot;
    int temp_less[LEN];
    int temp_more[LEN];
    int mid;
    int num_less = 0,num_more = 0;
    int i,j,k;
    pivot = a[start];
    for (i = start;i <= end;i++){
        if(a[i]<pivot){
            temp_less[num_less++] = a[i];
        }else{
            temp_more[num_more++] = a[i];
        }
    }
    mid = num_less;

    for(j = 0;j < num_less;j++){
        a[start+j] = temp_less[j];
    }
    for(k = 0;k < num_more;k++){
        a[mid+k] = temp_more[k];
    }

    // printf("mid = %d,num_less = %d,num_more = %d\n",mid,num_less,num_more);
    // printf("temp_less %d %d %d %d %d %d %d %d\n",temp_less[0], temp_less[1], temp_less[2], temp_less[3], \
    //  temp_less[4], temp_less[5], temp_less[6], temp_less[7]);
    // printf("temp_more %d %d %d %d %d %d %d %d\n",temp_more[0], temp_more[1], temp_more[2], temp_more[3], \
    //  temp_more[4], temp_more[5], temp_more[6], temp_more[7]);

    // printf("a         %d %d %d %d %d %d %d %d\n",a[0], a[1], a[2], a[3], a[4], a[5], a[6], a[7]);
	return mid;
}

void quicksort(int start, int end)
{
	int mid;
	if (end > start) {
		mid = partition(start, end);
		printf("mid (%d-%d, %d-%d) %d %d %d %d %d %d %d %d\n", 
		       start, mid, mid+1, end, 
		       a[0], a[1], a[2], a[3], a[4], a[5], a[6], a[7]);
		quicksort(start, mid-1);
		quicksort(mid+1, end);
		printf("quicksort (%d-%d, %d-%d) %d %d %d %d %d %d %d %d\n", 
		       start, mid, mid+1, end, 
		       a[0], a[1], a[2], a[3], a[4], a[5], a[6], a[7]);
	}
}

int main(){
    quicksort(0,LEN-1);
    return 0;
}
```
