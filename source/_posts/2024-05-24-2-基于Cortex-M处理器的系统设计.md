## 12 复位

需要同步复位异步释放：





>   // PORESET synchroniser
>   always @ (posedge FCLK or negedge PORESETn)
>     if (!PORESETn)
>       begin
>         poreset_n_q  <= 1'b0;
>         poreset_n_qq <= 1'b0;
>       end
>     else
>       begin
>         poreset_n_q  <= 1'b1;
>         poreset_n_qq <= poreset_n_q;
>       end
>
>   // SYSRESET synchroniser
>   always @ (posedge FCLK or negedge SYSRESETn)
>     if (!SYSRESETn)

>         sysreset_n_q  <= 1'b0;
>         sysreset_n_qq <= 1'b0;
>       end
>     else
>       begin
>         sysreset_n_q  <= 1'b1;
>         sysreset_n_qq <= sysreset_n_q;
>       end
>
